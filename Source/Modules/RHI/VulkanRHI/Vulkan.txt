option (USE_VULKAN "USE VULKAN" ON)
message (STATUS "USE_VULKAN ${USE_VULKAN}")

if (USE_VULKAN)
	set (VULKAN_DIRECTORY "Vulkan" CACHE STRING  "Vulkan directory")
	set (VULKAN_ARCH "msvc2019_64" CACHE STRING  "Vulkan arch")
	set (VULKAN_VERSION "1.3.280.0" CACHE STRING "Vulkan version")
	
	if (CMAKE_SYSTEM_NAME STREQUAL Windows)
		set (Vulkan_LIBS vulkan-1.lib)
		set (Vulkan_LINK_PATH		${LIBRARY_PREFIX_PATH}/${CMAKE_SYSTEM_NAME}/${VULKAN_DIRECTORY}/${VULKAN_VERSION}/${VULKAN_ARCH}/lib)
		set (Vulkan_INCLUDE_PATH 	${LIBRARY_PREFIX_PATH}/${CMAKE_SYSTEM_NAME}/${VULKAN_DIRECTORY}/${VULKAN_VERSION}/${VULKAN_ARCH}/include)
	#	set (Vulkan_INCLUDE_PATH	${Vulkan_INCLUDE_PATH} ${LIBRARY_PREFIX_PATH}/${CMAKE_SYSTEM_NAME}/${VULKAN_DIRECTORY}/${VULKAN_VERSION}/${VULKAN_ARCH}/include/vulkan)
		
	else ()
		set (Vulkan_LIBS opencv_core opencv_imgcodecs opencv_imgproc opencv_highgui opencv_video)
		set (Vulkan_LINK_PATH	/usr/local/lib)
		set (Vulkan_INCLUDE_PATH	${LIBRARY_PREFIX_PATH}/include/opencv /usr/local/include/OpenCV)
	endif ()
	message (STATUS "Vulkan_LIBS ${Vulkan_LIBS}")
	message (STATUS "Vulkan_LINK_PATH ${Vulkan_LINK_PATH}")
	message (STATUS "Vulkan_INCLUDE_PATH ${Vulkan_INCLUDE_PATH}")
endif ()
